/* Sample program in Tiny language – computes factorial*/
int main()
{
    int x;
    read x; /*input an integer*/
    if x > 0 /*don’t compute if x <= 0 */
    then 
        int fact := 1;

        repeat
            fact := fact * x;
            x := x - 1;
        until x = 0

        write fact; /*output factorial of x*/
    end
    return 0;
}
